Clustering

library(ggplot2)
irisCluster <- kmeans(iris[, 3:4], 3, nstart = 20)
irisCluster$cluster <- as.factor(irisCluster$cluster)
ggplot(iris, aes(Petal.Length, Petal.Width, color = irisCluster$cluster)) + geom_point()
